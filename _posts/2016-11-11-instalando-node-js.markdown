* sudo apt-get update.  

* apt-get install build-essential libssl-dev  

* sudo apt-get install curl  

* curl https://raw.githubusercontent.com/creationix/nvm/v0.16.1/install.sh | sh  

* nvm  

* nvm ls-remote  

* nvm install 6.9.1  

* node -v  

* nvm alias default 6.9.1  

* node  
  
* cd .nvm/v6.9.1/lib/  

* npm install npm@3.3.2
